template<typename T>
void printType(T value) {
    if constexpr (std::is_integral<T>::value)
        std::cout << "Integral\n";
    else if constexpr (std::is_floating_point<T>::value)
        std::cout << "Floating point\n";
    else
        std::cout << "Other type\n";
}
